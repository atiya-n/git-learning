
command for checking git version

    >git version

command for listing config detail

    >git config --list

command for adding user email

    >git config --global user.email "example@gmail.com"

command for adding the user name
    >git config --global user.name "eg-name"

command for initializing git repository

    >git init

command for ading all untracked files to staging area
    >git add -A 

command for adding staged files to local repository
    >git commit -m "message"

command for listing all untracked files
    >git status

command for getting the name that we saved for the remote repo (i.e. to check if we added the remote repository)
    >git remote -v 

command for adding the remote repo
    >git remote add name name_path

command to push the remote repo
    >git push name name_branch

command to get the name of the name_branch
    >git branch

command to check how many commits
    >git log

command to change the name of remote repo in saved in terminal
    >git remote rename old_name new_name

file to not track some files or folders
    > .gitignore
