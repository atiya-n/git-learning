
command for checking git version

    >git version

command for listing config detail

    >git config --list

command for adding user email

    >git config --global user.email "example@gmail.com"

command for initializing git repository

    >git init

command for ading all untracked files to staging area
    >git add -A 

command for adding staged files to local repository
    >git commit -m "message"

command for listing all untracked files
    >git status
